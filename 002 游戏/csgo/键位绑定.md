#游戏/CSGO/游戏设置/键位 

---

```CSGO
//清血迹

bind v "r_cleardecals";

//重丢上一个投掷物

bind c "sv_rethrow_last_grenade";

//换左右手

bind n "toggle cl_righthand 1 0";

//放bot

bind p "bot_place";

//沙鹰电击枪全甲闪

bind z "buy taser;buy deagle;buy vesthelm;buy flashbang;buy defuser";

//飞翔

bind x "noclip";

//发刀

bind l "say !drop";

//跑图参数

bind , "exec paotu.cfg"

//静音

bind h "clutch_mode_toggle";

//选队

bind m "slot12";

//电击枪快捷键

bind "MOUSE5" "slot11";

//一键大跳

alias +cjump "+jump; +duck";

alias -cjump "-jump; -duck";

bind space "+cjump";

//一键跳投

alias +jumpthrow "+jump;-attack";

alias -jumpthrow "-jump";

bind "MOUSE4“ +jumpthrow;
```
